/*
 * 样例输入：s = "3[a2[c]]"
 * 样例输出：accaccacc
 */
function fn (str) {
  let result = "";
  let num = 0;
  let subStr = "";
  if (!str.includes("[")) {
    return str;
  }
  str.split("[")[0].split("").forEach(i => {
    if (/\d/.test(i)) {
      num = Number(i);
    } else if (/[a-z]/.test(i)) {
      subStr = i;
    } else {
      num = 1
    }
  })

  for (let i = 0; i < num; i ++) {
    const newStr = str.replace(String(num), "").replace(subStr, "").replace("[", "").replace("]", "")
    result += fn(newStr)
  }

  return subStr + result;
}
