用栈：遇到左括号入栈；遇右括号与栈顶匹配（不匹配则无效）；遍历完栈空则有效。

```jsx
function isValid(s) {
  const map = { ')':'(', ']':'[', '}':'{' }, stack = [];
  for (const c of s) {
    if (!map[c]) stack.push(c);
    else if (stack.pop() !== map[c]) return false;
  }
  return stack.length === 0;
}
```
